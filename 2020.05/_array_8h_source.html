219 template <class T>
220 void Array<T>::setCapacity(unsigned int newCapacity)
221 {
222  // If the call does not come from the constructor
223  if (capacity_ != 0)
224  {
225  // Setting a new capacity is disabled if the array is fixed
226  if (fixedCapacity_)
227  {
228  LOGW_X("Trying to change the capacity of a fixed array, from from %u to %u", capacity_, newCapacity);
229  return;
230  }
231 
232  if (newCapacity == capacity_)
233  {
234  LOGW_X("Array capacity already equal to %u", capacity_);
235  return;
236  }
237  else if (newCapacity < capacity_)
238  LOGI_X("Array capacity shrinking from %u to %u", capacity_, newCapacity);
239  else if (newCapacity > capacity_)
240  LOGD_X("Array capacity growing from %u to %u", capacity_, newCapacity);
241  }
242 
243  T *newArray = nullptr;
244  if (newCapacity > 0)
245  newArray = new T[newCapacity];
246 
247  if (size_ > 0)
248  {
249  if (newCapacity < size_) // shrinking
250  size_ = newCapacity; // cropping last elements
251 
252  for (unsigned int i = 0; i < size_; i++)
253  newArray[i] = nctl::move(array_[i]);
254  }
255 
256  delete[] array_;
257  array_ = newArray;
258  capacity_ = newCapacity;
259 }

296 template <class T>
297 T *Array<T>::insertRange(unsigned int index, const T *firstPtr, const T *lastPtr)
298 {
299  // Cannot insert at more than one position after the last element
300  FATAL_ASSERT_MSG_X(index <= size_, "Index %u is out of bounds (size: %u)", index, size_);
301  FATAL_ASSERT_MSG_X(firstPtr <= lastPtr, "First pointer %p should precede or be equal to the last one %p", firstPtr, lastPtr);
302 
303  const unsigned int numElements = static_cast<unsigned int>(lastPtr - firstPtr);
304 
305  if (size_ + numElements > capacity_)
306  setCapacity((size_ + numElements) * 2);
307 
308  // Backwards loop to account for overlapping areas
309  for (unsigned int i = size_ - index; i > 0; i--)
310  array_[index + numElements + i - 1] = nctl::move(array_[index + i - 1]);
311  for (unsigned int i = 0; i < numElements; i++)
312  array_[index + i] = firstPtr[i];
313  size_ += numElements;
314 
315  return (array_ + index + numElements);
316 }
