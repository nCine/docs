368template <class K, class T, unsigned int Capacity, class HashFunc>
369bool StaticHashMap<K, T, Capacity, HashFunc>::remove(const K &key)
370{
371 if (size_ == 0)
372 return false;
373
374 const hash_t hash = hashFunc_(key);
375 const ProbeResult r = probe(key, hash);
376
377 if (!r.foundFlag)
378 return false;
379
380 unsigned int foundBucketIndex = r.found;
381 unsigned int prevFoundBucketIndex = r.prev;
382 unsigned int bucketIndex = foundBucketIndex;
383
384 // The found bucket is the last of the chain, previous one needs a delta fix
385 if (foundBucketIndex != hashes_[foundBucketIndex] % Capacity && delta2_[foundBucketIndex] == 0)
386 {
387 if (addDelta1(prevFoundBucketIndex) == foundBucketIndex)
388 delta1_[prevFoundBucketIndex] = 0;
389 else if (addDelta2(prevFoundBucketIndex) == foundBucketIndex)
390 delta2_[prevFoundBucketIndex] = 0;
391 }
392
393 while (delta1_[bucketIndex] != 0 || delta2_[bucketIndex] != 0)
394 {
395 unsigned int lastBucketIndex = bucketIndex;
396 if (delta1_[lastBucketIndex] != 0)
397 lastBucketIndex = addDelta1(lastBucketIndex);
398 if (delta2_[lastBucketIndex] != 0)
399 {
400 unsigned int secondLastBucketIndex = lastBucketIndex;
401 while (delta2_[lastBucketIndex] != 0)
402 {
403 secondLastBucketIndex = lastBucketIndex;
404 lastBucketIndex = addDelta2(lastBucketIndex);
405 }
406 delta2_[secondLastBucketIndex] = 0;
407 }
408 else
409 delta1_[bucketIndex] = 0;
410
411 if (bucketIndex != lastBucketIndex)
412 {
413 nodes_[bucketIndex].key = nctl::move(nodes_[lastBucketIndex].key);
414 nodes_[bucketIndex].value = nctl::move(nodes_[lastBucketIndex].value);
415 hashes_[bucketIndex] = hashes_[lastBucketIndex];
416 }
417
418 bucketIndex = lastBucketIndex;
419 }
420
421 hashes_[bucketIndex] = NullHash;
422 destructObject(nodes_ + bucketIndex);
423 size_--;
424
425 return true;
426}

453template <class K, class T, unsigned int Capacity, class HashFunc>
454typename StaticHashMap<K, T, Capacity, HashFunc>::ProbeResult
455StaticHashMap<K, T, Capacity, HashFunc>::probe(const K &key, hash_t hash) const
456{
457 ProbeResult r{};
458 r.ideal = hash % Capacity;
459 r.prev = r.ideal;
460
461 unsigned int bucketIndex = r.ideal;
462
463 if (bucketFoundOrEmpty(bucketIndex, hash, key) == false)
464 {
465 if (delta1_[bucketIndex] != 0)
466 {
467 bucketIndex = addDelta1(bucketIndex);
468 if (bucketFound(bucketIndex, hash, key) == false)
469 {
470 while (delta2_[bucketIndex] != 0)
471 {
472 bucketIndex = addDelta2(bucketIndex);
473 // Found at ideal index + delta1 + (n * delta2)
474 if (bucketFound(bucketIndex, hash, key))
475 {
476 r.found = bucketIndex;
477 r.foundFlag = true;
478 return r;
479 }
480 }
481
482 // At this point, bucketIndex is the tail
483 r.prev = bucketIndex; // for `patchDeltas()` to work
484
485 // Adding at ideal index + delta1 + (n * delta2)
486 r.empty = linearSearch(bucketIndex + 1, hash, key);
487 r.foundFlag = false;
488 return r;
489 }
490 else
491 {
492 // Found at ideal index + delta1
493 r.found = bucketIndex;
494 r.foundFlag = true;
495 return r;
496 }
497 }
498 else
499 {
500 // Adding at ideal index + delta1
501 r.empty = linearSearch(bucketIndex + 1, hash, key);
502 r.foundFlag = false;
503 return r;
504 }
505 }
506 else
507 {
508 // Using the ideal bucket index for the node
509 if (hashes_[bucketIndex] == NullHash)
510 {
511 r.empty = bucketIndex;
512 r.foundFlag = false;
513 return r;
514 }
515 else
516 {
517 r.found = bucketIndex;
518 r.foundFlag = true;
519 return r;
520 }
521 }
522}

524template <class K, class T, unsigned int Capacity, class HashFunc>
525unsigned int StaticHashMap<K, T, Capacity, HashFunc>::patchDeltas(const ProbeResult &r)
526{
527 if (r.ideal == r.empty)
528 return r.empty;
529
530 if (delta1_[r.ideal] == 0)
531 delta1_[r.ideal] = calcNewDelta(r.ideal, r.empty);
532 else // r.prev must be the tail
533 delta2_[r.prev] = calcNewDelta(r.prev, r.empty);
534
535 return r.empty;
536}
