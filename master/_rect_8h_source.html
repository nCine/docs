391template <class T>
392inline bool Rect<T>::overlaps(const Rect &rect) const
393{
394 const Vector2<T> rectMin = rect.min();
395 const Vector2<T> rectMax = rect.max();
396 const Vector2<T> thisMin = min();
397 const Vector2<T> thisMax = max();
398
399 const bool disjoint = (rectMax.x < thisMin.x || rectMin.x > thisMax.x ||
400 rectMax.y < thisMin.y || rectMin.y > thisMax.y);
401 return !disjoint;
402}

404template <class T>
405inline void Rect<T>::intersect(const Rect &rect)
406{
407 const Vector2<T> rectMin = rect.min();
408 const Vector2<T> rectMax = rect.max();
409
410 Vector2<T> newMin = min();
411 Vector2<T> newMax = max();
412 if (rectMin.x > newMin.x)
413 newMin.x = rectMin.x;
414 if (rectMin.y > newMin.y)
415 newMin.y = rectMin.y;
416 if (rectMax.x < newMax.x)
417 newMax.x = rectMax.x;
418 if (rectMax.y < newMax.y)
419 newMax.y = rectMax.y;
420
421 if (w < T(0))
422 nctl::swap(newMin.x, newMax.x);
423 if (h < T(0))
424 nctl::swap(newMin.y, newMax.y);
425
426 setMinMax(newMin, newMax);
427}
