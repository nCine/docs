390template <class T>
391T *Array<T>::insertRange(unsigned int index, const T *firstPtr, const T *lastPtr)
392{
393 // Cannot insert at more than one position after the last element
394 FATAL_ASSERT_MSG_X(index <= size_, "Index %u is out of bounds (size: %u)", index, size_);
395 FATAL_ASSERT_MSG_X(firstPtr <= lastPtr, "First pointer %p should precede or be equal to the last one %p", firstPtr, lastPtr);
396
397 const unsigned int numElements = static_cast<unsigned int>(lastPtr - firstPtr);
398 if (numElements == 0)
399 return (array_ + index);
400
401 if (size_ + numElements > capacity_)
402 setCapacity((size_ + numElements) * 2);
403
404 // Backwards loop to account for overlapping areas
405 for (unsigned int i = size_ - index; i > 0; i--)
406 {
407 T *src = &array_[index + i - 1];
408 T *dst = &array_[index + numElements + i - 1];
409
410 moveConstructObject(dst, src);
411 destructObject(src);
412 }
413 copyConstructArray(array_ + index, firstPtr, numElements);
414 size_ += numElements;
415
416 return (array_ + index + numElements);
417}

646template <class Tag>
647void Array<T>::setCapacityImpl(unsigned int newCapacity, Tag)
648{
649 // If the call does not come from the constructor
650 if (capacity_ != 0)
651 {
652 if (newCapacity == capacity_)
653 {
654 LOGW_X("Array capacity already equal to %u", capacity_);
655 return;
656 }
657 else
658 {
659 // Setting a new capacity is disabled if the array is fixed
660 FATAL_ASSERT_MSG_X(fixedCapacity_ == false, "Trying to change the capacity of a fixed array, from %u to %u", capacity_, newCapacity);
661
662 if (newCapacity == capacity_)
663 {
664 LOGW_X("Array capacity already equal to %u", capacity_);
665 return;
666 }
667 else if (newCapacity < capacity_)
668 LOGI_X("Array capacity shrinking from %u to %u", capacity_, newCapacity);
669 else // (newCapacity > capacity_)
670 LOGD_X("Array capacity growing from %u to %u", capacity_, newCapacity);
671 }
672 }
673
674 T *newArray = nullptr;
675 if (newCapacity > 0)
676 {
677#if !NCINE_WITH_ALLOCATORS
678 newArray = static_cast<T *>(::operator new(newCapacity * sizeof(T)));
679#else
680 newArray = static_cast<T *>(alloc_.allocate(newCapacity * sizeof(T)));
681#endif
682 }
683
684 if (size_ > 0)
685 {
686 const unsigned int oldSize = size_;
687 if (newCapacity < size_) // shrinking
688 size_ = newCapacity; // cropping last elements
689
690 moveConstructArray(newArray, array_, size_);
691 destructArray(array_, oldSize);
692 }
693
694#if !NCINE_WITH_ALLOCATORS
695 ::operator delete(array_);
696#else
697 alloc_.deallocate(array_);
698#endif
699 array_ = newArray;
700 capacity_ = newCapacity;
701}
