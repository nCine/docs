229template <class T, unsigned int C>
230void StaticArray<T, C>::setSize(unsigned int newSize)
231{
232 const int newElements = newSize - size_;
233
234 if (newSize > capacity_)
235 {
236 LOGW_X("Trying to extend the size of a static array beyond its cpacity, from from %u to %u", capacity_, newSize);
237 return;
238 }
239
240 if (newElements > 0)
241 constructArray(array_ + size_, newElements);
242 else if (newElements < 0)
243 destructArray(array_ + size_ + newElements, -newElements);
244 size_ += newElements;
245}

298template <class T, unsigned int C>
299T *StaticArray<T, C>::insertRange(unsigned int index, const T *firstPtr, const T *lastPtr)
300{
301 // Cannot insert at more than one position after the last element
302 FATAL_ASSERT_MSG_X(index <= size_, "Index %u is out of bounds (size: %u)", index, size_);
303 FATAL_ASSERT_MSG_X(firstPtr <= lastPtr, "First pointer %p should precede or be equal to the last one %p", firstPtr, lastPtr);
304
305 const unsigned int numElements = static_cast<unsigned int>(lastPtr - firstPtr);
306 if (numElements == 0)
307 return (array_ + index);
308 FATAL_ASSERT_MSG_X(size_ + numElements <= capacity_, "Can't add element beyond capacity (%d)", capacity_);
309
310 // Backwards loop to account for overlapping areas
311 for (unsigned int i = size_ - index; i > 0; i--)
312 {
313 T *src = &array_[index + i - 1];
314 T *dst = &array_[index + numElements + i - 1];
315
316 moveConstructObject(dst, src);
317 destructObject(src);
318 }
319 copyConstructArray(array_ + index, firstPtr, numElements);
320 size_ += numElements;
321
322 return (array_ + index + numElements);
323}

440{
441 // Cannot remove past the last element

443 FATAL_ASSERT_MSG_X(lastIndex <= size_, "Last index %u out of size range", lastIndex);

447 moveAssignArray(array_ + firstIndex, array_ + lastIndex, size_ - lastIndex);
