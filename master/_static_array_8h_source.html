301 else if (newElements < 0)

357template <class T, unsigned int Capacity>
358T *StaticArray<T, Capacity>::insertRange(unsigned int index, const T *firstPtr, const T *lastPtr)
359{
360 // Cannot insert at more than one position after the last element
361 FATAL_ASSERT_MSG_X(index <= size_, "Index %u is out of bounds (size: %u)", index, size_);
362 FATAL_ASSERT_MSG_X(firstPtr <= lastPtr, "First pointer %p should precede or be equal to the last one %p", firstPtr, lastPtr);
363
364 const unsigned int numElements = static_cast<unsigned int>(lastPtr - firstPtr);
365 if (numElements == 0)
366 return (array_ + index);
367 FATAL_ASSERT_MSG_X(size_ + numElements <= Capacity, "Can't add element beyond capacity (%d)", Capacity);
368
369 // Backwards loop to account for overlapping areas
370 for (unsigned int i = size_ - index; i > 0; i--)
371 {
372 T *src = &array_[index + i - 1];
373 T *dst = &array_[index + numElements + i - 1];
374
375 moveConstructObject(dst, src);
376 destructObject(src);
377 }
378 copyConstructArray(array_ + index, firstPtr, numElements);
379 size_ += numElements;
380
381 return (array_ + index + numElements);
382}

388 const unsigned int MaxInsertions = Capacity - size_;
389 if (end - initList.begin() > MaxInsertions)
390 end = initList.begin() + MaxInsertions;
391
392 return insertRange(index, initList.begin(), end);
