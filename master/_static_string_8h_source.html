317template <unsigned int Capacity>
318unsigned int StaticString<Capacity>::replace(const StaticString &source, unsigned int srcChar, unsigned int numChar, unsigned int destChar)
319{
320 // Clamping parameters to string lengths and capacities
321
322 // Cannot copy from beyond the end of the source string
323 const unsigned int clampedSrcChar = min(srcChar, source.length_);
324 const char *srcStart = source.array_ + clampedSrcChar;
325 // It is possible to write beyond the end of the destination string, but without creating holes
326 const unsigned int clampedDestChar = min(destChar, length_);
327
328 // Cannot copy more characters than the source has left until its length or more than the destination has until its capacity
329 const unsigned int charsToCopy = min(min(numChar, source.length_ - clampedSrcChar), Capacity - clampedDestChar - 1);
330
331 if (charsToCopy > 0)
332 {
333 char *destStart = array_ + clampedDestChar;
334
335 nctl::strncpy(destStart, Capacity - clampedDestChar, srcStart, charsToCopy);
336 // Destination string length can only grow, truncation has to be performed by the calling function using the return value
337 length_ = max(length_, static_cast<unsigned int>(destStart - array_) + charsToCopy);
338 array_[length_] = '\0';
339 }
340
341 return charsToCopy;
342}
