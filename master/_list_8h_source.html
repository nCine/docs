457 template <class T>
458 void List<T>::splice(Iterator position, List &source, Iterator first, Iterator last)
459 {
460  // Early-out if the source list is empty
461  if (source.isEmpty())
462  return;
463 
464  BaseListNode *node = position.node_;
465  BaseListNode *firstNode = first.node_;
466 
467  BaseListNode *firstPrev = firstNode->previous_;
468  BaseListNode *lastIncludedNode = first.node_;
469  while (first != last)
470  {
471  lastIncludedNode = first.node_;
472 
473  ++first;
474  size_++;
475  source.size_--;
476  }
477 
478  // it also works if `firstPrev` is the source sentinel
479  firstPrev->next_ = lastIncludedNode->next_;
480  // it also works if `lastIncludedNode` is the source sentinel
481  lastIncludedNode->next_->previous_ = firstPrev;
482 
483  BaseListNode *prevNode = node->previous_;
484  node->previous_ = lastIncludedNode;
485  firstNode->previous_ = prevNode;
486  lastIncludedNode->next_ = node;
487  // it also works if `prevNode` is the sentinel
488  prevNode->next_ = firstNode;

490  lastIncludedNode->next_ = node;
