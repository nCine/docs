287template <class T>
288inline Matrix4x4<T> Quaternion<T>::toMatrix4x4() const
289{
290 const T x2 = x * 2;
291 const T y2 = y * 2;
292 const T z2 = z * 2;
293
294 const T xx = x * x2;
295 const T xy = x * y2;
296 const T xz = x * z2;
297 const T yy = y * y2;
298 const T yz = y * z2;
299 const T zz = z * z2;
300
301 const T xw = w * x2;
302 const T yw = w * y2;
303 const T zw = w * z2;
304
305 return Matrix4x4<T>(Vector4<T>(1 - (yy + zz), xy + zw, xz - yw, 0),
306 Vector4<T>(xy - zw, 1 - (xx + zz), yz + xw, 0),
307 Vector4<T>(xz + yw, yz - xw, 1 - (xx + yy), 0),
308 Vector4<T>(0, 0, 0, 1));
309}
