313bool SparseSet<T>::insert(T value)
314{
315 if (value > maxValue_ || size_ >= capacity_ || contains(value) == true)
316 return false;
317
318 dense_[size_] = value;
319 sparse_[value] = size_;
320 size_++;

326template <class T>
327bool SparseSet<T>::insert(std::initializer_list<T> initList)
328{
329 const unsigned int MaxInsertions = capacity() - size();
330
331 unsigned int numInserted = 0;
332 for (const T element : initList)
333 {
334 if (numInserted >= MaxInsertions)
335 break;
336 const bool inserted = insert(element);
337 numInserted += inserted ? 1 : 0;
338 }
339
340 return (numInserted == initList.size());
341}

369bool SparseSet<T>::remove(T value)
370{
371 if (contains(value) == false)
372 return false;
373
374 T temp = dense_[size_ - 1];
375 dense_[sparse_[value]] = temp;
376 sparse_[temp] = sparse_[value];
